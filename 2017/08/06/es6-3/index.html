函数的扩展之箭头函数(es6-3)

作者:李华炎    发布于: 2017-08-06

  • 函数参数的默认值
function sum (a, b) {
  b = b || 2;    //平常以这种方式设置参数的默认值
  return a + b;
}
console.log(sum(5));    //使用默认值 7
console.log(sum(5,10)); //15

//或
function sum (a, b=2) {    //设置默认值
  return a + b;
}
  • 箭头函数
           //参数    //返回值
var sum = (a, b=2) => a+b;        
console.log(sum(5));    //b没有实参，使用默认值2 //结果 7
console.log(sum(5,10)); //15

// 或
var sum = (a, b=2) => {
  console.log("如果有多行代码就使用大括号");
  return a+b;        
}
  • 箭头函数中的this
    • 箭头函数的this始终指向函数定义时的this
var obj = {
  a : 10,
  func : function () {
    console.log(this.a);
  },
  test : function () {
    setTimeout(function () {
      this.func();        // 这里报错，因为this===window
    },100);
  }
}
obj.test();


//  改进 -将this赋值给其他变量
 var obj = {
   a : 10,
   func : function () {
     console.log(this.a);
   },
   test : function () {
     var that = this;
     setTimeout(function () {
       that.func(); // this === window 所以才需要这步var that = this;将this保存到that中
     },100);
   }
 }
 obj.test();  //10


// 改箭头函数
 var obj = {
   a : 10,
   func : function () {
     console.log(this.a);
   },
   test : function () {
     // test函数内部的this === obj
     setTimeout(() => {        // 箭头函数定义时this是test函数内部的this
       this.func();
     },100);
   }
 }
 obj.test();  //10

() => value;        //value指返回的值
